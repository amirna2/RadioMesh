112 explicit RadioMeshPacket(const std::vector<byte>& buffer)
113 {
114 // Simple sequential field parsing
115 protocolVersion = buffer[VERSION_POS];
116
117 // IDs (4 bytes each)
118
119 // use array copy for better performance
120 std::copy_n(buffer.begin() + SDEV_ID_POS, DEV_ID_LENGTH, sourceDevId.begin());
121 std::copy_n(buffer.begin() + DDEV_ID_POS, DEV_ID_LENGTH, destDevId.begin());
122 std::copy_n(buffer.begin() + PKT_ID_POS, MSG_ID_LENGTH, packetId.begin());
123
124 // Single byte fields
125 topic = buffer[TOPIC_POS];
126 deviceType = buffer[DEVICE_TYPE_POS];
127 hopCount = buffer[HOP_COUNT_POS];
128
129 // Multi-byte fields
130 packetCrc = (buffer[DATA_CRC_POS] << 24) | (buffer[DATA_CRC_POS + 1] << 16) |
131 (buffer[DATA_CRC_POS + 2] << 8) | buffer[DATA_CRC_POS + 3];
132
133 fcounter = (buffer[FCOUNTER_POS] << 24) | (buffer[FCOUNTER_POS + 1] << 16) |
134 (buffer[FCOUNTER_POS + 2] << 8) | buffer[FCOUNTER_POS + 3];
135
136 // Routing IDs
137 std::copy_n(buffer.begin() + LAST_HOP_ID_POS, DEV_ID_LENGTH, lastHopId.begin());
138 std::copy_n(buffer.begin() + NEXT_HOP_POS, DEV_ID_LENGTH, nextHopId.begin());
139 // Reserved and data
140 std::copy_n(buffer.begin() + RESERVED_POS, RESERVED_LENGTH, reserved.begin());
141 packetData.assign(buffer.begin() + DATA_POS, buffer.end());
142 }

148 std::vector<byte> toByteBuffer() const
149 {
150 std::vector<byte> buffer;
151 buffer.reserve(HEADER_LENGTH + packetData.size());
152
153 // Add header fields
154 buffer.push_back(protocolVersion);
155 buffer.insert(buffer.end(), sourceDevId.begin(), sourceDevId.end());
156 buffer.insert(buffer.end(), destDevId.begin(), destDevId.end());
157 buffer.insert(buffer.end(), packetId.begin(), packetId.end());
158 buffer.push_back(topic);
159 buffer.push_back(deviceType);
160 buffer.push_back(hopCount);
161
162 buffer.push_back((packetCrc >> 24) & 0xFF);
163 buffer.push_back((packetCrc >> 16) & 0xFF);
164 buffer.push_back((packetCrc >> 8) & 0xFF);
165 buffer.push_back(packetCrc & 0xFF);
166
167 buffer.push_back((fcounter >> 24) & 0xFF);
168 buffer.push_back((fcounter >> 16) & 0xFF);
169 buffer.push_back((fcounter >> 8) & 0xFF);
170 buffer.push_back(fcounter & 0xFF);
171
172 buffer.insert(buffer.end(), lastHopId.begin(), lastHopId.end());
173 buffer.insert(buffer.end(), nextHopId.begin(), nextHopId.end());
174 buffer.insert(buffer.end(), reserved.begin(), reserved.end());
175
176 // Add data section
177 buffer.insert(buffer.end(), packetData.begin(), packetData.end());
178
179 return buffer;
180 }

275 bool extractMIC(std::array<byte, MIC_LENGTH>& mic) const
276 {
277 if (!shouldUseMIC() || packetData.size() < MIC_LENGTH) {

279 }
