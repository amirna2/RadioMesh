111 explicit RadioMeshPacket(const std::vector<byte>& buffer)
112 {
113 // Simple sequential field parsing
114 protocolVersion = buffer[VERSION_POS];
115
116 // IDs (4 bytes each)
117
118 // use array copy for better performance
119 std::copy_n(buffer.begin() + SDEV_ID_POS, DEV_ID_LENGTH, sourceDevId.begin());
120 std::copy_n(buffer.begin() + DDEV_ID_POS, DEV_ID_LENGTH, destDevId.begin());
121 std::copy_n(buffer.begin() + PKT_ID_POS, MSG_ID_LENGTH, packetId.begin());
122
123 // Single byte fields
124 topic = buffer[TOPIC_POS];
125 deviceType = buffer[DEVICE_TYPE_POS];
126 hopCount = buffer[HOP_COUNT_POS];
127
128 // Multi-byte fields
129 packetCrc = (buffer[DATA_CRC_POS] << 24) | (buffer[DATA_CRC_POS + 1] << 16) |
130 (buffer[DATA_CRC_POS + 2] << 8) | buffer[DATA_CRC_POS + 3];
131
132 fcounter = (buffer[FCOUNTER_POS] << 24) | (buffer[FCOUNTER_POS + 1] << 16) |
133 (buffer[FCOUNTER_POS + 2] << 8) | buffer[FCOUNTER_POS + 3];
134
135 // Routing IDs
136 std::copy_n(buffer.begin() + LAST_HOP_ID_POS, DEV_ID_LENGTH, lastHopId.begin());
137 std::copy_n(buffer.begin() + NEXT_HOP_POS, DEV_ID_LENGTH, nextHopId.begin());
138 // Reserved and data
139 std::copy_n(buffer.begin() + RESERVED_POS, RESERVED_LENGTH, reserved.begin());
140 packetData.assign(buffer.begin() + DATA_POS, buffer.end());
141 }

147 std::vector<byte> toByteBuffer() const
148 {
149 std::vector<byte> buffer;
150 buffer.reserve(HEADER_LENGTH + packetData.size());
151
152 // Add header fields
153 buffer.push_back(protocolVersion);
154 buffer.insert(buffer.end(), sourceDevId.begin(), sourceDevId.end());
155 buffer.insert(buffer.end(), destDevId.begin(), destDevId.end());
156 buffer.insert(buffer.end(), packetId.begin(), packetId.end());
157 buffer.push_back(topic);
158 buffer.push_back(deviceType);
159 buffer.push_back(hopCount);
160
161 buffer.push_back((packetCrc >> 24) & 0xFF);
162 buffer.push_back((packetCrc >> 16) & 0xFF);
163 buffer.push_back((packetCrc >> 8) & 0xFF);
164 buffer.push_back(packetCrc & 0xFF);
165
166 buffer.push_back((fcounter >> 24) & 0xFF);
167 buffer.push_back((fcounter >> 16) & 0xFF);
168 buffer.push_back((fcounter >> 8) & 0xFF);
169 buffer.push_back(fcounter & 0xFF);
170
171 buffer.insert(buffer.end(), lastHopId.begin(), lastHopId.end());
172 buffer.insert(buffer.end(), nextHopId.begin(), nextHopId.end());
173 buffer.insert(buffer.end(), reserved.begin(), reserved.end());
174
175 // Add data section
176 buffer.insert(buffer.end(), packetData.begin(), packetData.end());
177
178 return buffer;
179 }
